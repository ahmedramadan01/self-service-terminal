192  q.put(root)
193  while not q.qsize() == 0:
194  menu = q.get()
195  export_list.append(menu)
196  for submenu in Menu.objects.filter(parent_menu=menu.pk):
197  if submenu not in discovered:
198  discovered.append(submenu)
199  q.put(submenu)

202  settings = Terminal_Settings.objects.get(title='settings')
203  forms = list(Form.objects.all())

207  "menus": list(),
208  "settings": list(),
209  "forms": list()
210  }
211

213  for menu in export_list:
214  menu = vars(menu)

217
